70 void EventFile::addDrop(const std::string& rule) {
71 
72  if (parent_ == 0)
73  return;
74 
75  size_t i = rule.find("keep");
76  bool iskeep = i != std::string::npos;
77  if (!iskeep)
78  i = rule.find("drop");
79 
80  if (i == std::string::npos)
81  return;
82 
83  std::string srule = rule.substr(i + 4);
84  for (i = srule.find_first_of(" \t\n\r"); i != std::string::npos; i = srule.find_first_of(" \t\n\r"))
85  srule.erase(i, 1);
86 
87  if (srule.length() == 0)
88  return;
89 
90  if (srule.back() != '*')
91  srule += '*';
92 
93  parent_->tree_->SetBranchStatus(srule.c_str(), (iskeep) ? (1) : (0));
94 }

96 bool EventFile::nextEvent() {
97 
98  if (ientry_ < 0 && parent_) {
99  if (!parent_->tree_) {
100  EXCEPTION_RAISE("EventFile", "No event tree in the file");
101  }
102  tree_ = parent_->tree_->CloneTree(0);
103  event_->setInputTree(parent_->tree_);
104  event_->setOutputTree(tree_);
105 
106  // Fill map of run numbers to RunHeader objects.
107  createRunMap();
108  }
109 
110  // close up the last event
111  if (ientry_ >= 0) {
112  if (isOutputFile_) {
113  event_->beforeFill();
114  tree_->Fill(); // fill the clones...
115  event_->Clear();
116  }
117  if (event_) {
118  event_->onEndOfEvent();
119  }
120  }
121 
122  if (parent_) {
123  if (!parent_->nextEvent()) {
124  return false;
125  }
126  parent_->tree_->GetEntry(parent_->ientry_);
127  ientry_ = parent_->ientry_;
128  event_->nextEvent();
129  entries_++;
130  return true;
131 
132  } else {
133 
134  // if we are reading, move the pointer
135  if (!isOutputFile_) {
136 
137  if (ientry_ + 1 >= entries_) {
138  return false;
139  }
140 
141  ientry_++;
142  tree_->LoadTree(ientry_);
143 
144  if (event_) {
145  event_->nextEvent();
146  }
147  return true;
148 
149  } else {
150  ientry_++;
151  entries_++;
152  return true;
153  }
154  }
155  return false;
